106    lattice_system: str | None = None,
107    centering: str | None = None,
108) -> Lattice:
109    """
110    Create a Lattice object.
111
112    Parameters
113    ----------
114    lattice_system: lattice system
115
116    centering: lattice centering
117
118    Return value
119    ------------
120    lattice: Lattice object with the specified `lattice_system` and `centering`
121    """
122    # --- Check arguments
123
124    # ------ `lattice_system`
125
126    if not isinstance(lattice_system, str):
127        raise ValueError(
128            f"`lattice_system` must be a string. (lattice_system={lattice_system})"
129        )
130
131    # Enforce that `lattice_system` is lowercase
132    lattice_system = lattice_system.lower()
133
134    # ------ `centering`
135
136    if not isinstance(centering, str):
137        raise ValueError(f"`centering` must be a string. (centering={centering})")
138
139    # Enforce that `centering` is lowercase
140    centering = centering.lower()
141
142    # --- Construct Lattice
143
144    try:
145        lattice = Lattice(LatticeSystem(lattice_system), Centering(centering))
146
147    except ValueError as error:
148        if "is not a valid LatticeSystem" in str(error):
149            raise ValueError(
150                "`lattice_system` must be one of "
151                f"{LatticeSystem.values()}. (lattice_system='{lattice_system}')."
152            )
153
154        if "is not a valid Centering" in str(error):
155            raise ValueError(
156                "`centering` must be one of "
157                f"{Centering.values()}. (centering='{centering}')."
158            )
159
160    return lattice

163def standardize_lattice(lattice: Lattice) -> Lattice:
164    """
165    Standardize and validate field values for `lattice`.
166
167    Parameters
168    ----------
169    lattice: Lattice to standardize
170
171    Return value
172    ------------
173    lattice: standardized Lattice
174    """
175    # --- Validate types
176
177    for field, value in zip(lattice._fields, lattice):
178        if not isinstance(value, str):
179            raise ValueError(
180                f"`lattice.{field}` must be a string. (lattice.{field}={value})"
181            )
182
183    # --- Standardize field values
184
185    standardized_field_values = {}
186    for field, value in zip(lattice._fields, lattice):
187        standardized_field_values[field] = value.lower()
188
189    # --- Construct Lattice with standardized field values
190
191    lattice = create_lattice(
192        standardized_field_values["lattice_system"],
193        standardized_field_values["centering"],
194    )
195
196    return lattice
